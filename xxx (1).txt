	ColorSequenceKeypoint.new(0.0, Color3.fromRGB(255, 0, 0)),  
	ColorSequenceKeypoint.new(0.2, Color3.fromRGB(255, 170, 0)),  
	ColorSequenceKeypoint.new(0.4, Color3.fromRGB(0, 255, 0)),

local button = Instance.new("TextButton", frame)

Instance.new("UICorner", button).CornerRadius = UDim.new(0, 6)

-- üîÅ Danh s√°ch t·ªça ƒë·ªô m·ª•c ti√™u
local targetPositions = {
	Vector3.new(-285.59, 26.00, 215.25),
	Vector3.new(-285.97, 26.00, 0.50),
	Vector3.new(-285.52, 26.00, 109.85),
	Vector3.new(-285.70, 26.00, -110.77),
	Vector3.new(-535.85, 26.00, 221.78),
	Vector3.new(-536.00, 26.00, 16.44),
	Vector3.new(-535.74, 26.00, 113.77),
	Vector3.new(-536.07, 26.00, -100.17)
}

-- üîÅ Spam teleport khi nh·∫•n  
button.MouseButton1Click:Connect(function()  
	local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()  
	local hrp = char:FindFirstChild("HumanoidRootPart")  
	if not hrp then return end  
  
	local original = hrp.Position    
	local upPos = original + Vector3.new(0, 100, 0)

	-- üîç T√¨m h∆∞·ªõng g·∫ßn nh·∫•t t·ª´ upPos
	local closestPos = nil
	local shortestDist = math.huge
	for _, pos in ipairs(targetPositions) do
		local dist = (Vector3.new(pos.X, upPos.Y, pos.Z) - upPos).Magnitude
		if dist < shortestDist then
			shortestDist = dist
			closestPos = pos
		end
	end

	-- T√≠nh h∆∞·ªõng v√† d·ªãch ra 100 studs
	local direction = (Vector3.new(closestPos.X, upPos.Y, closestPos.Z) - upPos).Unit
	local moveDir = upPos + direction * 100

	local start = tick()    
	while tick() - start < 3 do    
		-- Teleport l√™n tr·ªùi 100 + h∆∞·ªõng
		hrp.CFrame = CFrame.new(moveDir)    
		wait(0.002)    

		-- Teleport xu·ªëng c·ª±c s√¢u    
		hrp.CFrame = CFrame.new(0, 0 , 50000000000000000)    
		wait(0.2)    

		-- Quay l·∫°i tr√™n tr·ªùi
		hrp.CFrame = CFrame.new(moveDir)    
	end  
end)
